def min_number_of_months(df, min_months=12):
    """At least 6 months of data."""
    cond = df.groupby('user_id').ym.nunique() >= min_months
    users = cond[cond].index
    return df[df.user_id.isin(users)]

def add_variables(df):
    """Create helper variables."""
    y = df.transaction_date.dt.year.to_numpy('U')
    m = df.transaction_date.dt.month.to_numpy('U')
    m = np.char.zfill(m, 2)
    ym = np.char.add(y, '-')
    df['ym'] = np.char.add(ym, m)
    df['ymn'] = np.char.add(y, m).astype(int)
    return df

def income_pmts(df, prop=3/4):
    """Income payments in prop of all observed months."""
    def helper(g):
        tot_months = g.ym.nunique()
        inc_months = g[g.tag.str.contains('_income')].ym.nunique()
        return (inc_months / tot_months) > (prop)
    data = df[['user_id', 'transaction_date', 'tag', 'ym']]
    usrs = data.groupby('user_id').filter(helper).user_id.unique()
    return df[df.user_id.isin(usrs)]

def mygroupby(df, idx, x):
    
    data = df[[idx, x]].values
    data = np.sort(data, axis=0)
    idx = data[:, 0]
    x = data[:, 1]

    group_boundaries = np.unique(idx, return_index=True)[1][1:]
    grouped = np.split(x, group_boundaries)
    
    return [np.sum(g) for g in grouped]
